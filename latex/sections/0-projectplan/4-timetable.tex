\label{sec:project_plan}
% Describe a timeline via a Gantt chart or table with achievements per week.
 %Describe how the available time is expected to be spent. It is often difficult to identify all activities in advance, let alone to indicate accurately on the day what will be completed on which date. In general, it is possible to identify a number of phases (eg literature research, design, implementation, experiments, writing a thesis, etc.) and to make a weekly schedule. Also consider the interim results, the dependencies that may exist and whether there are critical dependencies that prevent the project from proceeding and what the alternative plan is in that case. A Gantt or Pert chart is often used for this.

% Gantt Chart
% For more complex Gantt charts see documentation here: 
% http://mirror.ox.ac.uk/sites/ctan.org/graphics/pgf/contrib/pgfgantt/pgfgantt.pdf
\begin{ganttchart}[
    expand chart=0.9\linewidth,
    vgrid,
    hgrid
    ]{0}{11}
        % Titles
        \gantttitle{Weeks}{12} \\
        \gantttitlelist{1,...,12}{1} \\

        % Group
        \ganttgroup{Literature Research}{0}{2} \\  % elem 0
        % Concrete tasks
        \ganttbar{PDF research}{1}{3} \\  % elem 1
        % More groups, further tasks ommitted
        \ganttgroup{Design}{2}{2}\\
        \ganttmilestone{Finish Design}{2}{2} \ganttnewline  % elem 4
        \ganttgroup{Implementation}{3}{7} \\  % elem 5
        \ganttbar{Setup Experiments}{4}{4} \\
        \ganttgroup{Experimentation}{5}{7} \\  % elem 6
        \ganttmilestone{Finish Experiments}{8} \ganttnewline % elem 7
        \ganttgroup{Evaluation}{9}{9}\\
        \ganttbar{Writing}{0,1}{10} \\  % elem 8
        \ganttgroup{Buffer}{11}{11} % elem 9
        
        % Connectors
        \ganttlink{elem2}{elem3}
        \ganttlink{elem3}{elem4}
        \ganttlink{elem6}{elem7}
        \ganttlink{elem7}{elem8}
        \ganttlink{elem4}{elem7}
  \label{ganttchart}
\end{ganttchart}
